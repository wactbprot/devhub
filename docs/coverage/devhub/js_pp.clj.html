001  (ns devhub.js-pp
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles javascript post-processing."}
004    (:require [clojure.string         :as string]
005              [cheshire.core          :as che]
006              [clojure.java.shell     :refer [sh]]
007              [devhub.utils           :as u]
008              [clojure.java.io        :as io]            
009              [com.brunobonacci.mulog :as µ]))
010  
011  (defn pp-source
012    [pp data]
013    (when (and (map? data) (vector? pp) (:_x data))
014       (-> (string/join pp)
015           (string/replace (re-pattern "_x")       (che/encode (:_x       data)))
016           (string/replace (re-pattern "_t_start") (che/encode (:_t_start data)))
017           (string/replace (re-pattern "_t_stop")  (che/encode (:_t_stop  data))))))
018  
019  (defn pp-file [task] (str (u/tmp-folder) "/" (:TaskName task) ".js"))
020  (defn exec-fn [conf] (str (:js-path conf) "/" (:js-exec conf)))
021  
022  (defn exec
023    "Executes the js `:PostProcessing` (pp).
024  
025    Example:
026    ```shell
027    ;; (sh node (exec-fn conf) (:js-path conf) pf)
028    ;; means e.g.:
029    node resources/js/exec.js resources/js/ /tmp/MKT50-exec.js
030    ```"
031    ([{conf :post} task]
032     task)
033    ([{conf :post} task data]
034    (spit (pp-file task) (pp-source (:PostProcessing task) data))
035    (let [res (sh (:js conf) (exec-fn conf) (:js-path conf) (pp-file task))]
036      (if-not (zero? (:exit res)) {:error (:err res)}
037              (try (che/decode (:out res) true)
038                   (catch Exception e
039                     (µ/log ::exec :error "decode error" :req-id (:req-id task))
040                     {:error (str "caught exception: " (.getMessage e))}))))))
