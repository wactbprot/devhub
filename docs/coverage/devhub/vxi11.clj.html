006              [com.brunobonacci.mulog :as µ])
007    (:import  [jvxi11 VXI11Factory VXI11UserFactory]))

009  (defn query
010    "Sends the `cmds` to a vxi11 device. Data is read out by means of the
011    byte buffer `bb` with the size `bs`.
012  
013    TODO: Calculate and set TimeOut.
014    "
015    [conf task]
016    (let [{pa :PrimaryAddress sa :SecondaryAddress host :Host device :DeviceName} task
017          ctrl (VXI11Factory/create host device)
018          usr  (VXI11UserFactory/create)
019          _    (.connect ctrl usr)]
020      (let [dev-or-err (try (.createDevice ctrl pa sa)
021                            (catch Exception e
022                              {:error "can not connect to device"}))]
023        (if (:error dev-or-err)
024          dev-or-err
025          (let [dev  dev-or-err
026                _    (.connect dev usr)
027                bs   (:read-buffer-size conf)
028                bb   (byte-array bs) 
029                f    (fn [cmd]
030                       (.write dev usr (.getBytes cmd) (.length cmd))
031                       (when-not (:NoReply task) (.read dev usr bb bs))
032                       (u/bb->string bb))
033                data (u/run f conf task)]
034            (.disconnect dev)
035            data)))))
036  
037  (defn handler
038    "Handles VXI11 queries.
039    
040    Example:
041    ```clojure
042    (def c (u/config))
043    (handler c {:Device \"gpib0,8\" :Host \"e75465\" :Value \":meas:func\"})
044    ;; =>
045    ;; {:data
046    ;; {:_x [MEASURING  1.015],
047    ;; :_t_start [1607513740001],
048    ;; :_t_stop [1607513740028],
049    ;; :_dt [27]}}
050    ```"
051    [{conf :vxi} task]
052      (let [data-or-err (query conf task)]
053        (if (:error data-or-err)
054          (let [error   data-or-err
055                err-msg (:error error)]
056            (µ/log ::handler :error err-msg :req-id (:req-id task))
057            error)
058          (let [data (u/meas-vec data-or-err)]
059            (µ/log ::handler :data data  :req-id (:req-id task))
060            data))))
