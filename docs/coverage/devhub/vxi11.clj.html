001  (ns devhub.vxi11
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles VXI11 Actions."}
004    (:require [devhub.utils           :as u]
005              [devhub.safe            :as safe]
006              [com.brunobonacci.mulog :as µ])
007    (:import  [jvxi11 VXI11Factory VXI11UserFactory]))
008  
009  (defn query
010    "Handles VXI11 queries. Sends the `cmds` to a vxi11 device. Data is
011    read out by means of the byte buffer `bb` with the size `bs`.
012    
013    Example:
014    ```clojure
015    (def c (u/config))
016    (def t {:Host \"e75465\" :Value [\":meas:func\"] :Wait 10 :Repeat 2
017            :DeviceName \"gpib0\" :PrimaryAddress 8 :SecondaryAddress 0})
018    (query c t)
019    ;; =>
020    ;; [[{:_x MEASURING  2.006,
021    ;;  :_t_start 1609749524988,
022    ;;  :_t_stop 1609749525020,
023    ;;  :_dt 32}]
024    ;; [{:_x MEASURING  2.006,
025    ;;  :_t_start 1609749525031,
026    ;;  :_t_stop 1609749525064,
027    ;;  :_dt 33}]]
028    ```"
029    [{conf :vxi} task]
030    (let [{host :Host device :DeviceName pa :PrimaryAddress sa :SecondaryAddress} task]
031      (µ/log ::query :req-id (:req-id task) :Host host :DeviceName device)
032      (let [ctrl (VXI11Factory/create host device)
033            usr  (VXI11UserFactory/create)
034            _    (.connect ctrl usr)
035            dev  (.createDevice ctrl pa sa)
036            conn-or-err (try (.connect dev usr) 
037                             (catch Exception e
038                               (µ/log ::query :error "connection error" :req-id (:req-id task))
039                               {:error "can not connect"}))]
040        (if (:error conn-or-err) conn-or-err
041            (let [bs (:read-buffer-size conf)
042                  bb (byte-array bs) 
043                  f  (fn [cmd]
044                       (.write dev usr (.getBytes cmd) (.length cmd))
045                       (when-not (:NoReply task) (.read dev usr bb bs))
046                       (u/bb->string bb))
047                  data (u/run f conf task)]
048              (.disconnect dev)
049              data)))))
