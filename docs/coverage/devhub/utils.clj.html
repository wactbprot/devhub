057  (defn reshape
058    "Returns `data` if `data` is a map. Transforms `data` from a single
059    measurement to a measurement vector if the length is greater than 1.
060  
061    REVIEW: There is a better solution (better than 4 times
062    `mapv`). However, n=10 takes 0.217172 msecs
063  
064    Solution one (1.41838 msecs):
065    ```clojure
066    (reduce (fn [n o] (update-in
067                       (update-in
068                        (update-in
069                         (update-in n [:_t_stop] conj (:_t_stop o))
070                       [:_t_start] conj (:_t_start o))
071                      [:_x] conj (:_x o))
072                     [:_dt] conj (:_dt o)))
073          {:_x [] :_t_start [] :_t_stop []}
074          (flatten v))
075    ```"
076    [data]
077    (cond
078      (:error  data) data
079      (vector? data) (let [v (flatten data)]
080                       {:_x       (mapv :_x       v)
081                        :_t_start (mapv :_t_start v)
082                        :_t_stop  (mapv :_t_stop  v)}) 
083      (map?    data) data
084      (nil?    data) (let [msg "no data"]
085                       (mu/log ::reshape :error msg)
086                       {:error msg})))

128    (fn [cmd] (let [t0 (ms)] (timestamp {:_x (f cmd)} t0 (ms)))))

130  (defn run
131    "Calls the function `f` with  all commands in `cmds` (vector of
132    strings or int).`repeat`s (int) and `wait`s (int) in between if `(>
133    repeat 1)`."
134    [f conf task]
135    (let [{cmds :Value  w :Wait n :Repeat} task]
136    (if (= 1 n (count cmds))
137      ((wrap-log task (wrap-times f)) (first cmds))
138      (mapv (fn [_]
139              (let [v (mapv (wrap-log task (wrap-times f)) cmds)]
140                (Thread/sleep w)
141                v))
142            (range n)))))
