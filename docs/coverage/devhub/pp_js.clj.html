001  (ns devhub.pp-js
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles javascript post-processing."}
004    (:require [clojure.string         :as string]
005              [cheshire.core          :as che]
006              [clojure.java.shell     :refer [sh]]
007              [devhub.utils           :as u]
008              [clojure.java.io        :as io]            
009              [com.brunobonacci.mulog :as µ]))
010  
011  (defn source-file [task] (str (u/tmp-folder) "/" (:TaskName task) "-source.js"))
012  
013  (defn data-file [task] (str (u/tmp-folder) "/" (:TaskName task) "-data.js"))
014  
015  (defn exec-file [conf]  (str (:js-path conf) "/" (:js-exec conf)))
016  
017  (defn exec
018    "Executes the js `:PostProcessing` (pp).
019  
020    Example:
021    ```shell
022    ;; (sh node (exec-fn conf) (:js-path conf) pf)
023    ;; means e.g.:
024    node resources/js/exec.js resources/js/ /tmp/MKT50-exec-source.js /tmp/MKT50-exec-data.js
025    ```"
026   [{conf :post} task]
027    (let [sf  (source-file task) 
028          df  (data-file task)
029          _   (spit df (che/encode (u/data task)))
030          _   (spit sf (string/join (:PostProcessing task)))
031          res (sh (:js conf) (exec-file conf) (:js-path conf) sf df)]
032      (if-not (zero? (:exit res))
033        (let [msg (:err res)]
034           (µ/log ::exec :error msg :req-id (:req-id task))
035           (merge task {:error msg}))
036        (merge task (try
037                      (che/decode (:out res) true)
038                      (catch Exception e
039                        (let [msg (str "caught exception: " (.getMessage e))]
040                          (µ/log ::exec :error msg :req-id (:req-id task))
041                          {:error msg})))))))
