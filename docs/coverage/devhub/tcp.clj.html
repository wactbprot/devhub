001  (ns devhub.tcp
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles TCP Actions."}
004    (:require [devhub.utils           :as u]
005              [com.brunobonacci.mulog :as µ])
006    (:import [java.io BufferedReader OutputStreamWriter InputStreamReader PrintWriter]
007             [java.net Socket]))
008  
009  (defn query
010    "Handles TCP queries. Sends the `cmds` to a raw tcp socket with the
011    specified `host` and `port`.
012      
013    Example:
014    ```clojure
015    (def c (u/config))
016    (def t {:Port 5025 :Host \"e75496\" :Value [\"ch101()\\n\"] :Wait 10 :Repeat 2})
017    (query c t)
018    ```"
019    [{conf :tcp} task]
020    (let [{host :Host port :Port} task
021          data (try
022                 (µ/log ::query :req-id (:req-id task) :Host host :Port port)
023                 (with-open [sock (Socket. host port)
024                             out  (PrintWriter.    (OutputStreamWriter. (.getOutputStream sock)))
025                             in   (BufferedReader. (InputStreamReader. (.getInputStream sock)))]
026                   (u/run (fn [cmd]
027                            (.print out cmd)
028                            (.flush out)
029                            (if-not (:NoReply task) (.readLine in) "")) conf task))
030                 (catch  Exception e
031                   (let [msg "connection error, can not connect to host"]
032                     (µ/log ::query :error msg :req-id (:req-id task))
033                     {:error msg})))]
034      (merge task (u/reshape data))))
