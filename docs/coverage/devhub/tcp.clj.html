001  (ns devhub.tcp
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles TCP Actions."}
004    (:require [devhub.utils           :as u]
005              [com.brunobonacci.mulog :as mu])
006    (:import [java.io BufferedReader OutputStreamWriter InputStreamReader PrintWriter]
007             [java.net Socket]))
008  
009  (defn out-socket [s] (PrintWriter. (OutputStreamWriter. (.getOutputStream s))))
010  (defn in-socket [s] (BufferedReader. (InputStreamReader. (.getInputStream s))))
011  (defn gen-socket [{h :Host p :Port}] (Socket. h p))
012  
013  (defn query
014    "Sends the `cmds` given by the `:Value` to a raw tcp socket with the
015    specified `host` and `port`.
016  
017    Example:
018    ```clojure
019    (def c (u/config))
020    (def t {:Port 5025 :Host \"e75496\" :Value [\"room()\\n\"] :Wait 10 :Repeat 2})
021    ;; prologix mks670
022    (def t {:Port 1234 :Host \"192.168.98.204\" :Wait 10 :Repeat 10
023    :Value [\"++addr 2\r++auto 1\r++eot_char 10\r:meas:func\r\"]})
024    (query c t)
025    ```"
026    [{conf :tcp} task]
027    (if-not (u/connectable? task)
028      {:error "can not connect"}
029      (with-open [sock (gen-socket task)
030                  out  (out-socket sock)
031                  in   (in-socket sock)]
032        (let [f (fn [cmd]
033                  (when-not (empty? cmd)
034                    (.print out cmd)
035                    (.flush out)
036                    (Thread/sleep 10))
037                  (if-not (:NoReply task)
038                    (.readLine in)
039                    ""))]
040          (u/run f conf task)))))
041  
042  (comment
043    ;; read chars to debug result string
044    (loop []
045      (let [c (.read in)]
046        (when (> c 0)
047          (prn (char c))
048          (recur)))))
049  
050  (defn handler
051    "Handles TCP queries."
052    [conf {host :Host port :Port req-id :req-id error :error :as task}]
053    (mu/trace ::handler [:function "tcp/handler"]
054              (if error
055                task
056                (let [_ (mu/log ::handler :req-id req-id :Host host :Port port)
057                      data (query conf task)]
058                  (merge task (if (:error data)
059                                (let [msg (:error data)]
060                                  (mu/log ::handler :error msg :req-id req-id)
061                                  data)
062                                (let [msg "received data"]
063                                  (mu/log ::handler :message msg :req-id req-id)
064                                  (u/reshape data))))))))
