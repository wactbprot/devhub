001  (ns devhub.tcp
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles TCP Actions."}
004    (:require [devhub.utils           :as u]
005              [com.brunobonacci.mulog :as µ])
006    (:import [java.io BufferedReader OutputStreamWriter InputStreamReader PrintWriter]
007             [java.net Socket]))
008  
009  (defn out-socket [sock] (PrintWriter. (OutputStreamWriter. (.getOutputStream sock))))
010  (defn in-socket [sock] (BufferedReader. (InputStreamReader. (.getInputStream sock))))
011  (defn gen-socket [{h :Host p :Port}] (Socket. h p))
012  
013  (defn query
014    "Sends the `cmds` given by the `:Value` to a raw tcp socket with the
015    specified `host` and `port`.
016  
017    Example:
018    ```clojure
019    (def c (u/config))
020    (def t {:Port 5025 :Host \"e75496\" :Value [\"ch101()\\n\"] :Wait 10 :Repeat 2})
021    (query c t)
022    ```"
023    [conf task]
024    (if-not (u/connectable? task)
025      {:error "can not connect"}
026      (with-open [sock (gen-socket task)
027                     out  (out-socket sock)
028                  in   (in-socket sock)]
029        (let [f (fn [cmd]
030                  (.print out cmd)
031                     (.flush out)
032                  (if-not (:NoReply task) (.readLine in) ""))]
033          (u/run f conf task)))))
034  
035  (defn handler
036    "Handles TCP queries."
037    [{conf :tcp} task]
038    (if (:error task) task
039        (let [{host :Host port :Port req-id :req-id} task
040              _    (µ/log ::query :req-id req-id :Host host :Port port)
041              data (query conf task)]
042          (merge task (if (:error data)
043                        (let [msg (:error data)]
044                          (µ/log ::query :error msg :req-id req-id)
045                          data)
046                        (let [msg "received data"]
047                          (µ/log ::query :message msg :req-id req-id)
048                          (u/reshape data)))))))
