001  (ns devhub.tcp
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles TCP Actions."}
004    (:require [devhub.utils :as u]
005              [devhub.safe  :as safe]
006              [com.brunobonacci.mulog :as µ])
007    (:import [java.io BufferedReader OutputStreamWriter InputStreamReader PrintWriter]
008             [java.net Socket]))
009  
010  (defn query
011    "Sends the `cmds` to a raw tcp socket with the specified `host` and
012    `port`."
013    [conf task]
014    (let [{host :Host port :Port} task]
015      (try
016        (with-open [sock (Socket. host port)
017                    out  (PrintWriter.    (OutputStreamWriter. (.getOutputStream sock)))
018                    in   (BufferedReader. (InputStreamReader. (.getInputStream sock)))]
019          (u/run (fn [cmd]
020                   (.print out cmd)
021                   (.flush out)
022                   (if-not (:NoReply task) (.readLine in) "")) conf task))
023        (catch  Exception e
024        {:error "can not connect to host"}))))
025  
026  (defn handler
027    "Handles TCP queries.
028      
029    Example:
030    ```clojure
031    (def c (u/config))
032    ;;
033    (def t1 {:Port 5025 :Host \"e75496\" :Value \"frs()\\n\"})
034    (handler c t1)
035    ;;
036    (def t2 {:Port 5000 :Host \"localhost\" :Value \"frs()\\n\"})
037    (handler c t2)
038    ```"
039    [{conf :tcp} task]
040      (let [data-or-err (query conf task)]
041        (if (:error data-or-err)
042          (let [error   data-or-err
043                err-msg (:error error)]
044            (µ/log ::handler :error err-msg :req-id (:req-id task))
045            error)
046          (let [data (u/meas-vec data-or-err)]
047            (µ/log ::handler :data data  :req-id (:req-id task))
048            data))))
