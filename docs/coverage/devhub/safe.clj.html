027  (defn parse-gpib-str
028    "Returns a map with `:DeviceName` `:PrimaryAddress` and `:SecondaryAddress`.
029    
030    Example:
031    ```clojure
032    (parse-gpib-str \"gpib0,9\")
033    ;; =>
034    ;; {:DeviceName \"gpib0\", :PrimaryAddress 9, :SecondaryAddress 0}
035   
036    (parse-gpib-str \"gpib0,9,1\")
037    ;; =>
038    ;; {:DeviceName \"gpib0\", :PrimaryAddress 9, :SecondaryAddress 1}
039    ``` "
040    [s]
041    (when-let [v (re-find (re-matcher #"(gpib[0-9]*),([0-9]*),?([0-9]*)" s))]
042      {:DeviceName       (nth v 1)
043       :PrimaryAddress   (u/number (nth v 2))
044       :SecondaryAddress (if (= "" (nth v 3)) 0 (u/number (nth v 3)))}))

048  (defn parse-device-str
049    [s]
050    (when (string? s)
051      (cond
052        (string/starts-with? s "inst") (parse-inst-str s)
053        (string/starts-with? s "gpib") (parse-gpib-str s)
054        :else {:error "not a device string"})))

061      (if (and h v p)
062        (assoc task

064               :Wait    (wait conf w)
