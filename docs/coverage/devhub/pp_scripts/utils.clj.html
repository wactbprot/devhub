032  (defn operable
033    "Returns a vector of booleans indicating if the values are
034    usable in calculations.
035  
036    Example:
037    ```clojure
038    (operable [\"1\" 1.234E-5 0 \"a\" :number])
039    ;; =>
040    ;; [true true true false false]
041    (operable 1)
042    ;; =>
043    ;; nil
044    ```"
045    [v]
046    (when (seqable? v) (mapv (comp number? u/number) v)))

048  (defn operable-seq
049    "Shrinks the vector `v` down to operable values depending on vector `o`.
050  
051    Example:
052    ```clojure
053    (operable-seq [\"1\" 1.234E-5 0    \"a\"  :number]
054                  [true  true     true false  false])  
055    ;; =>
056    ;; [\"1\" 1.234E-5 0]
057    ```"
058    [v o]
059    (when (and (seqable? v) (seqable? o))
060      (mapv :value
061            (filter (fn [{ok? :take}] (when ok? :take))
062                    (mapv (fn [x y] {:take x :value y}) o v)))))
063    
064  (defn calc-seq
065    "Same as [[operable-seq]] but returns a vector of numbers.
066  
067    Example:
068    ```clojure
069      (calc-seq [\"1\" 1.234E-5 0    \"a\"  :number]
070                  [true  true     true false  false])  
071    ;; =>
072    ;; [1 1.234E-5 0]
073    ```"
074  
075    [v o]
076    (when (and (seqable? v) (seqable? o))
077      (mapv u/number (operable-seq v o))))

083  (defn sd
084    "Calculates the standard deviation of the vector `v`."
085    [v]
086    (when-let [mv (mean v)]
087      (Math/sqrt (/ 
088                  (reduce (fn [a b] (+ a (square (- b  mv)))) 0 v)
089                  (dec (count v))))))
