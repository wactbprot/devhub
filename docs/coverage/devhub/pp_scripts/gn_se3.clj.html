025  (defn anybus-float
026    "Converts `byte-array` to `float`.
027    
028    Example:
029    ```clojure
030    ;; approx. 2mbar:
031    (def  v [32825 23151 30976 257 32826 17559 34304 257 32826 6375 41472 257 ; 1T 
032            32832 161 9216 257 32832 120 12032 257 32832 161 9216 257
033            32832 338 40448 257 32832 202 6400 257 32832 338 40448 257
034            32831 65530 41216 257 32831 61434 63232 257 32831 65530 41216 257
035            32832 150 59136 257 32832 31 29184 257 32831 63482 52224 257 ; 5T..
036            32832 99 46336 257])
037    (anybus-float v 12 4)
038    ;; =>
039    ;; 2.0098352
040    (anybus-float v 12 4 \"kPa\")
041    ;; =>
042    ;; 0.20098352432250977  
043    ```"
044    ([v s n]
045     (ppu/vec->float (anybus-vec (subvec v s (+ s n)))))
046    ([v s n u]
047     (let [x (anybus-float v s n)]
048       (condp = (keyword (or u "mbar"))
049         :mbar  x
050         :Pa    (* x 100)
051         :kPa   (/ x 10)
052         (Âµ/log ::anybus-float :error "conversion not implemented")))))

054  (defn anybus-extract
055    [task start n u]
056    (if (u/single-meas? task)
057      (anybus-float (:_x task) start n u)
058      (mapv (fn [v] (anybus-float v start n u)) (:_x task))))

060  (defn anybus-readout
061    "Returns `Result` and `ToExchange` maps.
062    
063    NOTE: The anybus gateway is configured to deliver pressures in `mbar`."
064    [task]
065    (let [input (:PostScriptInput task)
066          pre   (:Prefix input) suf (:Suffix input) unit (:Unit input)
067          m     (:anybus-byte-start conf) n (:anybus-byte-count conf)
068          f     (fn [[dev-name start]]
069                  (ppu/vl-result (str pre dev-name suf)
070                                 (anybus-extract task start n unit)
071                                 unit)) 
072          res (mapv f m)]
073      (merge task
074             {:Result res
075              :ToExchange (into {} (map (fn [m] {(:Type m) m}) res))})))

077  (defn ok?
078    [p_current p_target]
079    (and (number? p_target)
080         (number? p_current) 
081         (> p_current  (* p_target  (- 1 0.02)))
082         (< p_current  (* p_target  (+ 1 0.02)))))
