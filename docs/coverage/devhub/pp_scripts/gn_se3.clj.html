007  (defn anybus-vec
008    "Returns a vector of 4 bytes that can be turned into a float.
009    ```clojure
010    (anybus-vec 32832,99,46336,257)
011    ;; =>
012    ;; [64 0 99 181]
013    ```"
014    [v]
015    [(ppu/b16->l (nth v 0))
016     (ppu/b16->h (nth v 1))
017     (ppu/b16->l (nth v 1))
018     (ppu/b16->h (nth v 2))])
019  
020  (defn anybus-float [v s n](ppu/vec->float (anybus-vec (subvec v s (+ s n)))))

022  (defn anybus-readout
023    "Returns `Result` and `ToExchange` maps.
024    
025    ```clojure
026    ;; approx. 200Pa:
027    (def m {:_x [32825,23151,30976,257,32826,17559,34304,257,32826,6375,41472,257,
028                 32832,161,9216,257,32832,120,12032,257,32832,161,9216,257,
029                 32832,338,40448,257,32832,202,6400,257,32832,338,40448,257,
030                 32831,65530,41216,257,32831,61434,63232,257,32831,65530,41216,257,
031                 32832,150,59136,257,32832,31,29184,257,32831,63482,52224,257,
032                 32832,99,46336,257]
033    :_t_start 2})
034    (anybus-readout conf m) 
035    ```"
036    [task data]
037    (let [m (:anybus-byte-start conf) n (:anybus-byte-count conf)
038          x (:_x data)
039          f (fn [[dev-name start]]
040              {dev-name (if (u/single-meas? data)
041                          (anybus-float x  start n)
042                          (mapv (fn [v] (anybus-float v start n)) x))})] 
043          
044      (mapv f m)))
