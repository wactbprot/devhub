008  (defn anybus-vec
009    "Returns a vector of 4 bytes that can be turned into a float.
010    ```clojure
011    (anybus-vec 32832 99 46336 257)
012    ;; =>
013    ;; [64 0 99 181]
014    ```"
015    [v]
016    [(ppu/b16->l (nth v 0))
017     (ppu/b16->h (nth v 1))
018     (ppu/b16->l (nth v 1))
019     (ppu/b16->h (nth v 2))])

021  (defn anybus-float
022    "Converts `byte-array` to `float`.
023    
024    Example:
025    ```clojure
026    ;; approx. 2mbar:
027    (def  v [32825 23151 30976 257 32826 17559 34304 257 32826 6375 41472 257 ; 1T 
028            32832 161 9216 257 32832 120 12032 257 32832 161 9216 257
029            32832 338 40448 257 32832 202 6400 257 32832 338 40448 257
030            32831 65530 41216 257 32831 61434 63232 257 32831 65530 41216 257
031            32832 150 59136 257 32832 31 29184 257 32831 63482 52224 257 ; 5T..
032            32832 99 46336 257])
033    (anybus-float v 12 4)
034    ;; =>
035    ;; 2.0098352
036    (anybus-float v 12 4 \"kPa\")
037    ;; =>
038    ;; 0.20098352432250977  
039    ```"
040    ([v s n]
041     (ppu/vec->float (anybus-vec (subvec v s (+ s n)))))
042    ([v s n u]
043     (let [x (anybus-float v s n)]
044       (condp = (keyword u)
045         :mbar  x
046         :Pa    (* x 100)
047         :kPa   (/ x 10)
048         (µ/log ::anybus-float :error "conversion not implemented")))))

050  (defn anybus-readout
051    "Returns `Result` and `ToExchange` maps.
052    
053    NOTE: The anybus gateway is configured to deliver pressures in `mbar`."
054    [task]
055    (let [input (:PostScriptInput task)
056          pre   (:Prefix input) suf (:Suffix input) unit (:Unit input)
057          m     (:anybus-byte-start conf) n (:anybus-byte-count conf)
058          f     (fn [[dev-name start]]
059                  (let [x (if (u/single-meas? task)
060                            (anybus-float (:_x task) start n)
061                            (mapv (fn [v] (anybus-float v start n)) (:_x task)))]
062                    (ppu/vl-result (str pre dev-name suf) x unit ))) 
063          res (mapv f m)]
064      (merge task {:Result res
065                   :ToExchange (into {} (map (fn [m] {(:Type m) m}) res))})))
