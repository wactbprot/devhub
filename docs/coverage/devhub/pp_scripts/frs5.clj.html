009  (defn lb-extract
010    [s]
011    (let [r #"^N\s*([-+])\s*([0-9]*\.[0-9]*)\s*lb\s*"
012          v (re-matches r s)]
013      (when (= (count v) 3)
014        (str (nth v 1) (nth v 2)))))
015  
016  (defn lb-read-out
017    [task]
018    (let [i (:PostScriptInput task)
019          v  (mapv lb-extract (rest (:_x task)))
020          o  (ppu/operable v)]
021      (merge task {:Result [(ppu/vl-result (:Type i) (ppu/calc-seq v o) "lb")]})))
