009  (defn extract
010    [s]
011    (let [r #"[PA]*\s*([0-9]{1}\.[0-9]{1,2}[E][-+][0-9]{2})"]
012    (second (re-matches r s))))

014  (defn read-out
015    "The controller returns the pressure in `mbar` even if the return
016    string looks like: `PA   5.07E-08`"
017    [task]
018    (let [input (:PostScriptInput task)
019          v     (mapv extract (:_x task))
020          o     (ppu/operable v)]
021      (ppu/vl-result (:Type input) (ppu/calc-seq v o) "mbar")))
022  
023  (defn pressure-rise
024    [task]
025    (let [input (:PostScriptInput task)
026          token (:Type input)
027          v     (mapv extract (:_x task))
028          o     (ppu/operable v)
029          y     (ppu/calc-seq v o)
030          t     (ppu/t0t1->t (ppu/calc-seq (:_t_start task) o)
031                             (ppu/calc-seq (:_t_stop task)  o))]
032      (merge task {:Result [(ppu/vl-result (str token "_slope_x")
033                                           (ppu/slope y t)
034                                           "mbar/ms")
035                            (ppu/vl-result (str token "_R")
036                                           (ppu/r-square y t)
037                                           "1")
038                            (ppu/vl-result (str token "_N")
039                                           (count y)
040                                           "1")]})))
