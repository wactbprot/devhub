001  (ns devhub.pp-scripts.vs_se3
002    (:require [devhub.pp-scripts.utils :as ppu]
003              [devhub.utils            :as u]))
004  
005  (def conf (u/config "vs_se3.edn"))
006  
007  (defn registers-ok?
008    "Checks `rs` for type vector and checks the length to
009    be `(:register-count conf)`."
010    [rs]
011    (and (vector? rs) (=  (count rs) (:register-count conf))))
012  
013  (defn check
014    "Returns a vector of maps.
015    
016    Example:
017    ```clojure
018    (check [89] {:ThingA [0 0] :ThingB [0 1]})
019    ;; =>
020    ;; [{:ThingA true} {:ThingB false}]
021    ```
022    "
023    [rs m]
024    (mapv
025     (fn [[kw [block position]]] {kw (ppu/open? (nth rs block) position)})
026     m))
027  
028  (defn valves
029    "Returns exchange structures like
030  
031    ```json
032    {ToExchange:
033      {E7_open:    {Bool: true},
034       E3_open:    {Bool: true},
035       E15_open:   {Bool: true},
036       E10_open:   {Bool: false},
037       E20_open:   {Bool: false},
038       E18_open:   {Bool: true},
039       E10_closed: {Bool: true},
040       ...
041       registers [89, 0, -102, 0, -102, 0, 21, 0, -91]}}
042    ```
043    
044    Example:
045    ```clojure
046    (valves {} {:_x [85 0 16 0 16 0 21 0 7]})
047    ```"
048    [task {rs :_x}]
049    (if (registers-ok? rs)
050      (let [vs (ppu/exch-bool-map (check rs (:valve-position conf)))]
051        {:ToExchange (reduce merge {:registers rs} vs)})
052      {:error "wrong register format"}))
053  
054  (defn switches
055    "Returns exchange structures like
056    
057    ```json
058    ToExchange:
059    {E1_open:{ Bool:true}, E1_closed:{Bool:false}
060    ```
061    
062    Example:
063    ```clojure
064    (switches {} {:_x [89 0 -102 0 -102 0 21 0 -91]})
065    ```"
066    [task {rs :_x}]
067    (if (registers-ok? rs)
068      (let [so (ppu/exch-bool-map (check rs (:switch-open    conf)))
069            sc (ppu/exch-bool-map (check rs (:switch-closed  conf)))]
070        {:ToExchange (reduce merge (reduce merge {:registers rs} so) sc)})
071      {:error "wrong register format"}))
