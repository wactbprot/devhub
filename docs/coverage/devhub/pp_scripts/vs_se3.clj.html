001  (ns devhub.pp-scripts.vs_se3
002    (:require [devhub.pp-utils :as ppu]
003              [devhub.utils    :as u]))
004  
005  (def conf (u/config "vs_se3.edn"))

007  (defn registers-ok?
008    "Checks `rs` for type vector and checks the length to
009    be `(:register-count conf)`."
010    [rs]
011    (and (vector? rs) (= (count rs) (:register-count conf))))
012  
013  (defn check
014    "Returns a vector of maps. The register `b`lock and
015    `p`osition is provided by the `conf`iguration.
016    
017    Example:
018    ```clojure
019    (check [89] {:ThingA [0 0] :ThingB [0 1]})
020    ;; =>
021    ;; [{:ThingA true} {:ThingB false}]
022    ```
023    "
024    [rs m]
025    (mapv (fn [[kw [b p]]] {kw (ppu/open? (nth rs b) p)}) m))
026  
027  (defn set-valve
028    "The PreScript vs_se3.set-valve calculates the new register value
029    depending on the current state. The current state is provided by the
030    registers PreInput. The `:Address`, register (`b`lock) and
031    `p`osition is provided by the `conf`iguration.
032   
033    Example:
034    ```clojure
035    (set-valve {:PreInput {
036                   :registers [1029 0 4100 0 1300 0 21248 0 83]
037                   :valve \"V1\"
038                   :should \"open\" }})
039    ```"
040    [task]
041    (let [input  (:PreInput task)
042          rs              (:registers input)
043          valve  (keyword (:valve     input))
044          should (keyword (:should    input))]
045      (if (and (registers-ok? rs) valve should)
046        (let [[b p] (valve (:valve-position conf))
047              a     (valve (:valve-set-addr conf))
048              r     (ppu/modify-register (nth rs b) p should)]
049          (assoc task :Value [r] :Address a))
050        (merge task {:error "missing or wrong :registers, :valve or :should"}))))
051  
052  (defn valves
053    "Returns the human readable state of the valves derived from the `reg`ister`s`.
054  
055    ```json
056    {ToExchange:
057      {E7_open:    {Bool: true},
058       E3_open:    {Bool: true},
059       E15_open:   {Bool: true},
060       E10_open:   {Bool: false},
061       E20_open:   {Bool: false},
062       E18_open:   {Bool: true},
063       E10_closed: {Bool: true},
064       ...
065       registers [89, 0, -102, 0, -102, 0, 21, 0, -91]}}
066    ```
067    
068    Example:
069    ```clojure
070    (get-valves {} {:_x [85 0 16 0 16 0 21 0 7]})
071    ```"
072    [task]
073    (let [rs (:_x task)]
074      (if (registers-ok? rs)
075        (let [vs (ppu/exch-bool-map (check rs (:valve-position conf)))]
076          (merge task {:ToExchange (reduce merge {:registers rs} vs)}))
077        (merge task {:error "wrong register format"}))))
078    
079  (defn switches
080    "Returns exchange structures like
081    
082    ```json
083    ToExchange:
084    {E1_open:{ Bool:true}, E1_closed:{Bool:false}
085    ```
086    
087    Example:
088    ```clojure
089    (switches {} {:_x [89 0 -102 0 -102 0 21 0 -91]})
090    ```"
091    [task]
092    (let [rs (:_x task)]
093      (if (registers-ok? rs)
094        (let [so (ppu/exch-bool-map (check rs (:switch-open    conf)))
095              sc (ppu/exch-bool-map (check rs (:switch-closed  conf)))]
096          (merge task {:ToExchange (reduce merge (reduce merge {:registers rs} so) sc)}))
097        (merge task {:error "wrong register format"}))))
