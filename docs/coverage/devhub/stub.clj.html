001  (ns devhub.stub
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles stub requests."}
004    (:require [devhub.utils :as u]
005              [devhub.safe :as safe]))
006  
007  (defn select-response
008    "Selects the response depending on the configuration. Implemented
009    methods are:
010    
011    * `:rand` (default)  
012    * `:first` (fallback)
013    * `:last`"
014    [kw rs mode]
015    (let [r  (kw rs)]
016      (condp = mode
017        :first (first r)
018        :last  (last  r)
019        :rand  (nth   r (rand-int (count r)))
020        (first r))))
021  
022  (defn all-responses [conf] (u/config (:response-file conf))) 
023   
024  (defn response
025    "Gets and returns a stub response if one is registered in `:stub-response-file`.
026  
027    Example:
028    ```clojure
029    (response (u/config) {:TaskName \"VS_SE3-get-valves-pos\"})
030    ```"
031    [{conf :stub} task]
032    (if-let [task (safe/stub conf task)]
033      (let [f (fn [_] (select-response (:select task) (all-responses conf) (:mode conf)))]
034        (if-let [data (u/run f conf task)]
035          (u/meas-vec data)
036          {:error true :reason "no data"}))
037      {:error "can not derive keyword fron task name"}))
