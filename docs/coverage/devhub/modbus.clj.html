001  (ns devhub.modbus
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles MODBUS Actions."}
004    (:require [devhub.utils           :as u]
005              [devhub.safe            :as safe]
006              [com.brunobonacci.mulog :as µ])
007    (:import [com.intelligt.modbus.jlibmodbus Modbus]
008             [com.intelligt.modbus.jlibmodbus.master ModbusMaster]
009             [com.intelligt.modbus.jlibmodbus.master ModbusMasterFactory]
010             [com.intelligt.modbus.jlibmodbus.tcp TcpParameters]
011             [java.net InetAddress]))
012  
013  (defn query
014    "Handles Modbus queries. Executes the query depending on the `FunctionCode`.
015  
016    Example:
017    ```clojure
018    (def mc (:modbus (u/config)))
019    (query mc {:Host \"e75446\" :Quantity 5 :Address 45407 :FunctionCode :ReadHoldingRegisters})
020  
021    (query mc {:Host \"e75480\" :Quantity 1 :Address 0 :FunctionCode :ReadInputRegisters})
022    ```"
023    [{conf :modbus} task]
024    (let [{host :Host fc :FunctionCode addr :Address q :Quantity} task]
025      (µ/log ::query :req-id (:req-id task) :Host host :Address addr)
026      (let [ok-or-err (try (InetAddress/getByName host)
027                           (catch  Exception e
028                             (µ/log ::query :error "connection error" :req-id (:req-id task))
029                             {:error "can not connect to host"}))]
030        (if (:error ok-or-err)
031          ok-or-err
032          (let [param  (TcpParameters. host (:port conf) (:keep-alive conf))
033                master (ModbusMasterFactory/createModbusMasterTCP param)
034                s-addr (:default-slave-address conf)]
035            (Modbus/setAutoIncrementTransactionId true)
036            (.connect master)
037            (let [f (condp = fc
038                    :ReadHoldingRegisters (fn [x] (.readHoldingRegisters master s-addr addr q)) 
039                    :ReadInputRegisters   (fn [x] (.readInputRegisters   master s-addr addr q))
040                    :ReadCoils            (fn [x] (.readCoils            master s-addr addr q)) 
041                    :ReadDiscreteInputs   (fn [x] (.readDiscreteInputs   master s-addr addr q))
042                    :writeSingleRegister  (fn [x] (.writeSingleRegister  master s-addr addr x)))
043                  data (u/run f conf task)]
044              (.disconnect master)
045              data))))))
