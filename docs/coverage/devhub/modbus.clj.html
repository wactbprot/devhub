001  (ns devhub.modbus
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles MODBUS Actions."}
004    (:require [devhub.utils           :as u]
005              [devhub.safe            :as safe]
006              [com.brunobonacci.mulog :as µ])
007    (:import [com.intelligt.modbus.jlibmodbus Modbus]
008             [com.intelligt.modbus.jlibmodbus.master ModbusMaster]
009             [com.intelligt.modbus.jlibmodbus.master ModbusMasterFactory]
010             [com.intelligt.modbus.jlibmodbus.tcp TcpParameters]))
011  
012  (defn I->v
013    "Returns a vector of integers made from the input `[I` type."
014    [I]
015    (mapv int I))
016  
017  (defn query
018    "Executes the query depending on the `FunctionCode`.
019    
020    Example:
021    ```clojure
022    (def c (u/config))
023    ;; valves
024    (def t {:Host \"e75446\" :Quantity 9 :Address 45407
025           :FunctionCode :ReadHoldingRegisters
026           :Value [:no-value] :Wait 10 :Repeat 1})
027    (query c t)
028    
029    ;; switches
030    (def t {:Host \"e75446\" :Quantity 9 :Address 45395
031           :FunctionCode :ReadHoldingRegisters
032           :Value [:no-value] :Wait 10 :Repeat 1})
033    (query c t)
034  
035    ;; cdgs 
036    (def t {:Host \"e75480\" :Quantity 64 :Address 0
037             :FunctionCode :ReadInputRegisters
038              :Value [:no-value] :Wait 10 :Repeat 1})
039    (query c t)
040    ```"
041    [conf task]
042    (if-not (u/connectable? task)
043      {:error "can not connect"}
044      (let [{host :Host fc :FunctionCode addr :Address q :Quantity} task
045            s-addr (:default-slave-address conf)
046            param  (TcpParameters. host (:port conf) (:keep-alive conf))
047            master (ModbusMasterFactory/createModbusMasterTCP param)]
048        (Modbus/setAutoIncrementTransactionId true)
049        (.connect master)
050        (let [f (condp = fc
051                  :ReadHoldingRegisters (fn [x] (I->v (.readHoldingRegisters master s-addr addr q))) 
052                  :ReadInputRegisters   (fn [x] (I->v (.readInputRegisters   master s-addr addr q)))
053                  :ReadCoils            (fn [x] (I->v (.readCoils            master s-addr addr q))) 
054                  :ReadDiscreteInputs   (fn [x] (I->v (.readDiscreteInputs   master s-addr addr q)))
055                  :writeSingleRegister  (fn [x] (.writeSingleRegister        master s-addr addr x)))
056              data (u/run f conf task)]
057          (.disconnect master)
058          data))))
059  
060  (defn handler
061    "Handles Modbus queries. "
062    [{conf :modbus} task]
063    (if (:error task) task
064        (let [{host :Host fc :FunctionCode addr :Address q :Quantity req-id :req-id} task
065              _    (µ/log ::query :req-id req-id :Host host :Address addr)
066              data (query conf task)]
067          (merge task (if (:error data)
068                        (let [msg (:error data)]
069                          (µ/log ::query :error msg :req-id req-id)
070                          data)
071                        (let [msg "received data"]
072                          (µ/log ::query :message msg :req-id req-id)
073                          (u/reshape data)))))))
