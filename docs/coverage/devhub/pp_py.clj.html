001  (ns devhub.pp-py
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles python post-processing."}
004    (:require [cheshire.core          :as che]
005              [clojure.java.shell     :refer [sh]]
006              [devhub.utils           :as u]
007              [com.brunobonacci.mulog :as µ]))
008  
009  (defn pp-file [conf task] (str (:PostScriptPy task)  ".py"))
010  
011  (defn exec
012    "Executes a predefined *python3* script given with the `:PostScriptPy` key.
013  
014    Example:
015    ```clojure
016    (def pc (u/config))
017    (def data {:_x \"a\\nba\\nb\\ncb\"})
018    (time
019      (exec pc {:PostScriptPy \"ls-demo\"} data))
020    ;; =>
021    ;; Elapsed time: 31.680355 msecs
022    ;; {:ToExchange {:FilesVector [a ba b cb]}}
023    ```"
024    [{conf :post} task]
025    (let [ps (pp-file conf task)]
026      (if-not (u/file? ps)
027        (let [msg (str "no such postscript file: " ps)]
028          (µ/log ::exec :error msg :req-id (:req-id task))
029          (merge task {:error msg}))
030        (let [res (sh (:py conf) ps (che/encode task) (che/encode (u/data task)) :dir (:py-path conf))]
031          (merge task (if-not (zero? (:exit res))
032                        {:error (:err res)}
033                        (try
034                          (che/decode (:out res) true)
035                          (catch Exception e
036                            (let [msg (str "decode error, caught exception: " (.getMessage e))]
037                              (µ/log ::exec :error msg :req-id (:req-id task))
038                              {:error msg})))))))))
