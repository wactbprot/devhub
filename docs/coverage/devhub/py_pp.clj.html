001  (ns devhub.py-pp
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Handles python post-processing."}
004    (:require [cheshire.core          :as che]
005              [clojure.java.shell     :refer [sh]]
006              [devhub.utils           :as u]
007              [com.brunobonacci.mulog :as µ]))
008  
009  (defn pp-file [conf task] (str (:PostScriptPy task)  ".py"))
010  
011  
012  (defn exec
013    "Executes a predefined *python3* script given with the `:PostScriptPy` key.
014  
015    Example:
016    ```clojure
017    (def pc (u/config))
018    (def data {:_x \"a\\nba\\nb\\ncb\"})
019    (time
020      (exec pc {:PostScriptPy \"ls-demo\"} data))
021    ;; =>
022    ;; Elapsed time: 31.680355 msecs
023    ;; {:ToExchange {:FilesVector [a ba b cb]}}
024    ```"
025    [{conf :post} task data]
026    (let [ps (pp-file conf task)]
027      (if-not (u/file? ps) {:error (str "no such postscript file: " ps)}
028              (let [res (sh (:py conf) ps (che/encode task) (che/encode data) :dir (:py-path conf))]
029                (if-not (= 0 (:exit res)) {:error (:err res)}
030                        (try (che/decode (:out res) true)
031                             (catch Exception e
032                               (µ/log ::exec :error "decode error" :req-id (:req-id task))
033                               {:error (str "decode error, caught exception: " (.getMessage e))})))))))
