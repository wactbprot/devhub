001  (ns devhub.py-pp
002    ^{:author "wactbprot"
003      :doc "Handles python post-processing."}
004    (:require [clojure.string     :as string]
005              [cheshire.core      :as che]
006              [clojure.java.shell :refer [sh]]
007              [devhub.utils       :as u]
008              [com.brunobonacci.mulog :as µ]))
009  
010  (defn pp-file [conf task] (str (:py-path conf)  "/" (:PostScriptPy task)  ".py"))
011  
012  (defn exec
013    "Executes a predefined *python3* script given with the `:PostScriptPy` key.
014  
015    Example:
016    ```clojure
017    (def pc (u/config))
018    (def data {:_x [\"a\\nb\" \"a\\nb\\nc\"]})
019    (time
020      (exec pc {:PostScriptPy \"ls-demo\"} data))
021    ;; =>
022    ;; Elapsed time: 116.616146 msecs
023    ;; {:ToExchange {:FileAmount [2 3]}}
024    ```"
025    [{conf :post} task data]
026    (let [ps (pp-file conf task)]
027      (if (u/file? ps)
028        (let [res (sh "python3" ps (che/encode data) (che/encode task))]
029          (if (= 0 (:exit res))
030            (try
031              (che/decode (:out res) true)
032              (catch Exception e
033                (µ/log ::exec :exception e :status :failed)
034                {:error (str "caught exception: " (.getMessage e))}))
035            {:error (:err res)}))
036        {:error (str "no such file: " ps)})))
