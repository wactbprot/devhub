034  (defn operable
035    "Returns a vector of booleans indicating if the values are
036    usable in calculations.
037  
038    Example:
039    ```clojure
040    (operable [\"1\" 1.234E-5 0 \"a\" :number])
041    ;; =>
042    ;; [true true true false false]
043    (operable 1)
044    ;; =>
045    ;; nil
046    ```"
047     [v]
048    (when (seqable? v) (mapv (comp number? u/number) v)))

050  (defn operable-seq
051    "Shrinks the vector `v` down to operable values depending on vector `o`.
052  
053    Example:
054    ```clojure
055    (operable-seq [\"1\" 1.234E-5 0    \"a\"  :number]
056                  [true  true     true false  false])  
057    ;; =>
058    ;; [\"1\" 1.234E-5 0]
059    ```"
060    [v o]
061    (when (and (seqable? v) (seqable? o))
062      (mapv :value
063            (filter (fn [{ok? :take}] (when ok? :take))
064                    (mapv (fn [x y] {:take x :value y}) o v)))))

101  (defn stdev
102    "Calculates the standard deviation of the vector `v`.
103  
104    Example:
105    ```clojure
106    (stdev [0 1 2])
107    ;; =>
108    ;; 1.0
109  
110    (stdev [0])
111    ;; =>
112    ;; nil
113  
114    (stdev [])
115    ;; =>
116    ;; nil
117  
118    (stdev nil)
119    ;; =>
120    ;; nil
121    ```"
122    [v]
123    (let [ndec (dec (count v)) mv (mean v)]
124      (when (and mv (pos? ndec))
125        (Math/sqrt (/ 
126                    (reduce (fn [a b] (+ a (square (- b  mv)))) 0 v)
127                    ndec)))))

131  (defn slope
132     "Ordinary Least Squares (OLS)"
133    [y x]
134    (let [mX    (mean x)
135          mY    (mean y)
136          x-mX  (mapv (fn [a] (- a mX)) x)
137          y-mY  (mapv (fn [a] (- a mY)) y)]
138      (/ (reduce + (mapv * x-mX y-mY))
139         (reduce + (mapv square x-mX))))) 
140  
141  (defn intercept [y x] (- (mean y) (* (slope y x) (mean x))))  
142  
143  (defn r-square
144    [y x]
145    (let [m     (slope y x)
146          c     (intercept y x)
147          mY    (mean y)
148          py    (mapv (fn [a] (+ c (* m a))) x)
149          y-mY  (mapv (fn [a] (- a mY)) y)
150          y-py  (mapv (fn [a b] (- a b)) y py)]
151      (- 1 (/ (reduce + (mapv square y-py))
152              (reduce + (mapv square y-mY))))))

154  (defn vl-result
155    "Returns the vl result structure at least consisting of `:Type` `t`
156    `:Value` `v` and `:Unit` `u`.
157  
158    Example:
159    ```clojure
160    (vl-result \"ind\" [1] \"Pa\")
161    ;; =>
162    ;; {:Type \"ind\", :Value 1, :Unit \"Pa\"}
163  
164    (vl-result \"ind\" 1 \"Pa\")
165    ;; =>
166    ;; {:Type \"ind\", :Value 1, :Unit \"Pa\"}
167  
168    (vl-result \"ind\" [0 1 2] \"Pa\")
169    ;; =>
170    ;; {:Type \"ind\", :Value 1, :Unit \"Pa\", :SdValue 1.0, :N 3}
171    ```"
172    [t v u]
173    (if (coll? v)
174      (let [m {:Type t :Value (mean v) :Unit u}]
175        (if-let [sd (stdev v)]
176          (assoc m :SdValue sd :N (count v))
177          m))
178      {:Type t :Value v :Unit u}))
