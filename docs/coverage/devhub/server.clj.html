024  ;;------------------------------------------------------------
025  ;; dispatch pre scripting or processing
026  ;;------------------------------------------------------------
027  (defn pre-dispatch
028    "Dispatches the pre-processing. The following processing paths are
029    implemented:
030  
031    * `:PreProcessing`: eval javascript strings
032    * `:PreScript`: clojure functions
033    * `:PreScriptPy`: python scripts
034  
035    The pre-processing returns the **task**."
036    [conf task]
037    (if (:error task) task
038        (cond
039          (:PreScript     task) (pp/pre-dispatch conf task)
040          (:PreProcessing task) (js/exec          conf task)
041          (:PreScriptPy   task) (py/exec          conf task)
042          :else (do (µ/log ::post-dispatch :req-id (:req-id task) :message "no pre-processing")
043                    task))))
044  
045  ;;------------------------------------------------------------
046  ;; dispatch post scripting or processing
047  ;;------------------------------------------------------------
048  (defn post-dispatch
049    "Dispatches the post-processing. The following processing paths are
050    implemented:
051  
052    * `:PostScript`: clojure functions
053    * `:PostProcessing`: javascript strings
054    * `:PostScriptPy`: python scripts
055  
056    The pre-processing returns the **data**."
057    [conf task]
058    (if (:error task) task
059        (cond
060          (:PostScript     task) (pp/post-dispatch  conf task)
061          (:PostProcessing task) (js/exec           conf task)
062          (:PostScriptPy   task) (py/exec           conf task)
063          :else (do
064                  (µ/log ::post-dispatch :req-id (:req-id task) :message "no post-processing")
065                  task))))
066  
067  ;;------------------------------------------------------------
068  ;; dispatch on action
069  ;;------------------------------------------------------------
070  (defmulti dispatch 
071    "Dispatches depending on the `:Action`. The following protocols paths are
072    implemented:
073  
074    * `:TCP`
075    * `:MODBUS`
076    * `:VXI11`
077    * `:EXECUTE`"  
078    (fn [conf task]
079      (if (:error task)
080        :error
081        (do (µ/log ::dispatch :req-id (:req-id task) :Action (:Action task))
082            (keyword (:Action task))))))
083    
084  (defmethod dispatch :error   [conf task] task)
085  (defmethod dispatch :TCP     [conf task] (tcp/handler     conf task))
086  (defmethod dispatch :MODBUS  [conf task] (modbus/handler  conf task))
087  (defmethod dispatch :VXI11   [conf task] (vxi/handler     conf task))
088  (defmethod dispatch :EXECUTE [conf task] (execute/handler conf task))
089  
090  (defmethod dispatch :default
091    [conf task]
092    (let [msg "wrong :Action"]
093      (µ/log ::dispatch :req-id (:req-id task) :error msg :Action (:Action task))
094      (merge task {:error msg})))
095  
096  ;;------------------------------------------------------------
097  ;; request thread
098  ;;------------------------------------------------------------
099  (defn thread 
100    [conf task stub?]
101    (let [task (safe/task conf task)]
102      (let [task (pre-dispatch conf task)]
103        (let [task (if stub? (stub/response conf task) (dispatch conf task))]
104          (let [task (sample/record conf task)]
105            (post-dispatch conf task))))))
106  
107  ;;------------------------------------------------------------
108  ;; routes
109  ;;------------------------------------------------------------
110  (defroutes app-routes
111    (µ/trace ::routes
112      [:function "routes"]
113      (POST "/stub"   [:as req] (res/response (thread (u/config) (u/task req) true)))
114      (POST "/"       [:as req] (res/response (thread (u/config) (u/task req) false)))
115      (POST "/echo"   [:as req] (res/response (u/task req)))
116      (GET "/version" [:as req] (res/response (u/version)))
117      (route/not-found "No such service.")))
118  
119  (def app
120    (-> (handler/site app-routes)
121        (middleware/wrap-json-body {:keywords? true})
122        middleware/wrap-json-response))

124  (defn init-log!
125    [{conf :mulog }]
126    (µ/set-global-context!
127     {:app-name "devhub" :version (:version (u/version))})
128    (µ/start-publisher! conf))

130  (def server (atom nil))
131  (def logger (atom nil))
132  
133  (defn stop
134    []
135    (µ/log ::stop)
136    (@server :timeout 100)
137    (reset! server nil)
138    (@logger)
139    (reset! logger nil))
140  
141  (defn start
142    ([]
143     (start (u/config)))
144    ([conf]
145     (µ/log ::start)
146     (reset! logger (init-log! conf))
147     (reset! server (run-server #'app (:server conf)))))
148  
149  (defn -main [& args] (u/ascii-logo)(start))
