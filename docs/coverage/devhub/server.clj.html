023  (defn pre-dispatch
024    "Dispatches the pre-processing. The following processing paths are
025    implemented:
026  
027    * `:PreProcessing`: eval javascript strings
028    * `:PreScript`: clojure functions
029    * `:PreScriptPy`: python scripts
030    
031    The pre-processing returns the **task**.
032    "
033    [conf task]
034    (let [{pp :PreProcessing
035           ps :PreScript
036           py :PreScriptPy} task]
037      (μ/log ::pre-dispatch :req-id (:req-id task) :PreProcessing pp :PreScript ps :PreScriptPy py)
038      (cond
039        pp (js/exec conf task)
040        ;; ps (clj-pp ps data)
041        :else task)))
042  
043  (defn post-dispatch
044    "Dispatches the post-processing. The following processing paths are
045    implemented:
046  
047    * `:PostProcessing`: javascript strings
048    * `:PostScript`: clojure functions
049    * `:PostScriptPy`: python scripts
050  
051    The pre-processing returns the **data**.
052    "
053    [conf task data]
054    (let [{pp :PostProcessing
055           ps :PostScript
056           py :PostScriptPy} task]
057      (μ/log ::post-dispatch :req-id (:req-id task)  :PostProcessing pp :PostScript ps :PostScriptPy py)
058      (cond
059        pp (js/exec      conf task data)
060        ps (clj/dispatch conf task data)
061        py (py/exec conf task data)
062        :else data)))
063  
064  (defn dispatch
065    "Dispatches depending on the `:Action`. The following protocols paths are
066    implemented:
067  
068    * `:TCP`
069    * `:MODBUS`
070    * `:VXI11`
071    * `:EXECUTE`
072    "     
073    [conf task]
074      (let [action (keyword (:Action task))]
075        (μ/log ::dispatch :req-id (:req-id task) :Action action)
076        (condp = action 
077          :TCP     (tcp/query       conf task)
078          :MODBUS  (modbus/query    conf task)
079          :VXI11   (vxi/query       conf task)
080          :EXECUTE (execute/handler conf task)
081          {:error "wrong action"})))
082  
083  (defn thread
084    [conf task stub?]
085    (μ/log ::thread :req-id (:req-id task) :stub stub? :task-name (:TaskName task))
086    (let [task (safe/task conf task)]
087      (if (:error task) task
088          (let [task (pre-dispatch conf task)]
089            (if (:error task) task
090                (let [data (if stub? (stub/response conf task) (dispatch conf task))]
091                  (if (:error data) data
092                      (post-dispatch conf task (u/meas-vec data)))))))))

094  (defroutes app-routes
095    (POST "/stub"   [:as req] (res/response (thread (u/config) (u/task req) true)))
096    (POST "/"       [:as req] (res/response (thread (u/config) (u/task req) false)))
097    (POST "/echo"   [:as req] (res/response (u/task req)))
098    (GET "/version" [:as req] (res/response (u/version)))
099    (route/not-found "No such service."))

116  (defn stop
117    []
118    (μ/log ::stop)
119    (@server :timeout 100)
120    (reset! server nil)
121    (@logger)
122    (reset! logger nil))

124  (defn start
125    ([]
126     (start (u/config)))
127    ([conf]
128     (μ/log ::start)
129     (reset! logger (init-log! conf))
130     (reset! server (run-server #'app (:server conf)))))
