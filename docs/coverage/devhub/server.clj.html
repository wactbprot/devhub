024  (defn pre-dispatch
025    "Dispatches the pre-processing. The following processing paths are
026    implemented:
027  
028    * `:PreProcessing`: eval javascript strings
029    * `:PreScript`: clojure functions
030    * `:PreScriptPy`: python scripts
031  
032    The pre-processing returns the **task**."
033    [conf task]
034    (cond
035      (:PreScript     task) (clj/pre-dispatch conf task)
036      (:PreProcessing task) (js/exec          conf task)
037      (:PreScriptPy   task) (py/exec          conf task)
038      :else (do
039              (μ/log ::post-dispatch :req-id (:req-id task) :message "no pre-processing")
040              task)))
041  
042  (defn post-dispatch
043    "Dispatches the post-processing. The following processing paths are
044    implemented:
045  
046    * `:PostProcessing`: javascript strings
047    * `:PostScript`: clojure functions
048    * `:PostScriptPy`: python scripts
049  
050    The pre-processing returns the **data**."
051    [conf task data]
052    (cond
053      (:PostScript     task) (clj/post-dispatch conf task data)
054      (:PostProcessing task) (js/exec           conf task data)
055      (:PostScriptPy   task) (py/exec           conf task data)
056      :else (do
057              (μ/log ::post-dispatch :req-id (:req-id task) :message "no post-processing")
058              data)))
059  
060  (defn dispatch
061    "Dispatches depending on the `:Action`. The following protocols paths are
062    implemented:
063  
064    * `:TCP`
065    * `:MODBUS`
066    * `:VXI11`
067    * `:EXECUTE`"
068    [conf task]
069      (let [action (keyword (:Action task))]
070        (μ/log ::dispatch :req-id (:req-id task) :Action action)
071        (condp = action
072          :TCP     (tcp/query       conf task)
073          :MODBUS  (modbus/query    conf task)
074          :VXI11   (vxi/query       conf task)
075          :EXECUTE (execute/handler conf task)
076          {:error "wrong action"})))
077  
078  (defn thread
079    [conf task stub?]
080    (μ/log ::thread :req-id (:req-id task) :stub stub? :task-name (:TaskName task))
081    (let [task (safe/task conf task)]
082      (if (:error task) task
083          (let [task (pre-dispatch conf task)]
084            (if (:error task) task
085                (let [data (if stub? (stub/response conf task) (dispatch conf task))]
086                  (if (:error data) data
087                      ;; (sample/record conf task (u/meas-vec data))
088                      (post-dispatch conf task (sample/record conf task (u/meas-vec data))))))))))

090  (defroutes app-routes
091    (POST "/stub"   [:as req] (res/response (thread (u/config) (u/task req) true)))
092    (POST "/"       [:as req] (res/response (thread (u/config) (u/task req) false)))
093    (POST "/echo"   [:as req] (res/response (u/task req)))
094    (GET "/version" [:as req] (res/response (u/version)))
095    (route/not-found "No such service."))

097  (def app
098    (-> (handler/site app-routes)
099        (middleware/wrap-json-body {:keywords? true})
100        middleware/wrap-json-response))

102  (defn init-log!
103    [{conf :mulog }]
104    (μ/set-global-context!
105     {:app-name "devhub" :version (:version (u/version)) :env "local"})
106    (μ/start-publisher! conf))

109  (def logger (atom nil))

111  (defn stop
112    []
113    (μ/log ::stop)
114    (@server :timeout 100)
115    (reset! server nil)
116    (@logger)
117    (reset! logger nil))
118  
119  (defn start
120    ([]
121     (start (u/config)))
122    ([conf]
123     (μ/log ::start)
124     (reset! logger (init-log! conf))
125     (reset! server (run-server #'app (:server conf)))))
