001  (ns devhub.sample
002    ^{:author "Wact B. Prot <wactbprot@gmail.com>"
003      :doc "Records samples from outer space (or not)."}
004    (:require [devhub.utils    :as u]
005              [clojure.java.io :as io]
006              [clojure.pprint :as pp]
007              [com.brunobonacci.mulog :as µ]))
008  
009  (defn insert
010    [{conf :sample} m kw data]
011    (let [n (:count conf) x (:_x data) t (:_t_start data)]
012      (assoc m kw (if (kw m)
013                    (into [] (take n (if (vector? t)
014                                       (concat (take n x) (kw m))
015                                       (conj (reverse (kw m)) x))))
016                    (if (vector? t) (into [] (take n x)) [x])))))
017  
018  (defn record
019    [conf {task-name :TaskName} data]
020    (when (and (u/record-sample? conf) task-name (:_x data))
021      (µ/log ::record :message "record sample data" :TaskName task-name)
022      (spit (io/resource (u/responses-file conf))
023            (with-out-str (pp/pprint (insert conf (u/all-responses conf) (keyword task-name) data)))))
024    data)
